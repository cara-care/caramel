Example

    <Slider
      minimum={0}
      maximum={10}
      onValueChange={(number) => {
        console.log(number);
      }}
      thumbWidth={50}
      showTooltipOnSlide
      leftText={'Low'}
      rightText={'High'}
      existingValue={5}
      thumbStyle={{
        height: 31,
        borderRadius: 15.5,
        borderColor: 'rgb(243, 243, 243)',
        borderWidth: 1,
        backgroundColor: 'white',
        shadowColor: '#000',
        shadowOffset: {
          width: 0,
          height: 2,
        },
        shadowOpacity: 0.3,
        shadowRadius: 3.0,
        elevation: 5,
      }}
    />
